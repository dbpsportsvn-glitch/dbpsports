# Cấu hình SMTP để gửi email thật
# Tạo file .env trong thư mục backend với nội dung sau:

SECRET_KEY=your-secret-key-here
DEBUG=True
ALLOWED_HOSTS=127.0.0.1,localhost

# Email Configuration (SMTP)
EMAIL_BACKEND=django.core.mail.backends.smtp.EmailBackend
EMAIL_HOST=smtp.gmail.com
EMAIL_PORT=587
EMAIL_HOST_USER=your-email@gmail.com
EMAIL_HOST_PASSWORD=your-app-password
EMAIL_USE_TLS=True
EMAIL_USE_SSL=False
DEFAULT_FROM_EMAIL=DBP Sports <your-email@gmail.com>

# Admin Email
ADMIN_EMAIL=admin@dbpsports.com

# Hướng dẫn:
# 1. Thay your-email@gmail.com bằng email Gmail của bạn
# 2. Tạo App Password cho Gmail:
#    - Vào Google Account Settings
#    - Security > 2-Step Verification > App passwords
#    - Tạo password mới cho "Mail"
#    - Thay your-app-password bằng password này
# 3. Lưu file với tên .env (không có extension)
# 4. Chạy: python send_email_real.py
